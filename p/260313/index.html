class Wrapper(torch.nn.Module):
    def __init__(self, model):
        super().__init__()
        self.model = model
    
    def forward(self, audio):
        features = self.model.extract_features(audio)
        # 特征值格式符合输出
        if isinstance(features, (tuple, list)):
            features = features[0]

        return features

wrapped_beats = Wrapper(model)

fun loadFromAsset(context: Context, assetName: String): BEATs {
    val modelFile = File(context.filesDir, assetName)

    if (!modelFile.exists()) {
        context.assets.open(assetName).use { input ->
            FileOutputStream(modelFile).use { output ->
                input.copyTo(output)
            }
        }
    }

    return loadFromFile(modelFile.absolutePath)
}

fun printInformation(text : String){
    val runtime : Runtime = Runtime.getRuntime();

    val javaMem : Long = runtime.totalMemory() - runtime.freeMemory();
    val nativeMem : Long = Debug.getNativeHeapAllocatedSize();

    Log.d("information", text)
    Log.d("information", (System.nanoTime()/1e6).toString() + "ms")
    Log.d("information","Java heap = " + javaMem/1024/1024 + " MB");
    Log.d("information","Native heap = " + nativeMem/1024/1024 + " MB");
}
